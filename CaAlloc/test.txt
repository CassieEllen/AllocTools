SET &THREE = 'THREE'
FILTLIST &INCONLY INCLUDE(ONE, 'TWO', THREE)
FILTLIST &INCEXC INCLUDE(ONE, 'TWO', THREE) EXCLUDE(THR*)
IF &THREE EQ &INCONLY THEN WRITE 'FOUND THREE' ELSE WRITE 'EXCLUDED THREE'
IF &THREE EQ &INCEXC THEN WRITE 'FOUND THREE' ELSE WRITE 'EXCLUDED THREE'

IF &THREE EQ 'THREE' THEN
WRITE 'EQUAL' 
ELSE
WRITE 'NOT EQUAL'

IF &THREE NE 'THREE' THEN
WRITE 'EQUAL' 
ELSE
WRITE 'NOT EQUAL'

IF &THREE EQ THREE THEN 
DO 
WRITE 'IN BLOCK'
WRITE 'EQUAL' 
END
ELSE
DO 
WRITE 'IN BLOCK'
WRITE 'NOT EQUAL' 
END

IF &THREE NE THREE THEN 
DO 
WRITE 'IN BLOCK'
WRITE 'EQUAL' 
END
ELSE
DO 
WRITE 'IN BLOCK'
WRITE 'NOT EQUAL' 
END
